CREATE SCHEMA `paradigmasJade` DEFAULT CHARACTER SET utf8 ;

CREATE TABLE `ARREMATANTE` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `NOME` varchar(45) DEFAULT NULL,
  `VALOR_DINHEIRO` decimal(10,0) DEFAULT NULL,
  PRIMARY KEY (`ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


  
  CREATE TABLE `LANCES` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `ID_ARREMATANTE` int(11) DEFAULT NULL,
  `ID_OBJETO` int(11) DEFAULT NULL,
  `VALOR_LANCE` decimal(10,0) DEFAULT NULL,
  PRIMARY KEY (`ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
  
  


  CREATE TABLE `OBJETO` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `NOME` varchar(45) DEFAULT NULL,
  `LANCE_INICIAL` decimal(10,0) DEFAULT NULL,
  `VALOR_INCREMENTO` decimal(10,0) DEFAULT NULL,
  PRIMARY KEY (`ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
